\begin{section}{Flow with upper and lower bound}
  \begin{multicols}{2}
    有源匯上下界最大流:
    \begin{enumerate}
      \item 建立附加網路
      \item 求max\_flow(S',T')
      \item 若有解，求max\_flow(S,T)
      \item nop
      \item nop
    \end{enumerate}
    有源匯上下界最小流
    \begin{enumerate}
      \item 建立附加網路(無(T,S)邊)
      \item 求max\_flow(S',T')
      \item 加入(T,S)邊
      \item 求max\_flow(S',T')
      \item 若S'、T'滿流，則(T,S)即為答案
    \end{enumerate}
  \end{multicols}
\end{section}
